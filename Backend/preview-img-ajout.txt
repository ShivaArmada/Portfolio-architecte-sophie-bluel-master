function updateVisualisation() {
        const preview_square = document.getElementsByClassName("preview_vehicle")[0];
        const imgInput = document.getElementById("images_vehicules_1");

        if (imgInput.files && imgInput.files[0]) {
            const reader = new FileReader();

            reader.onload = function(e) {
                const imageUrl = e.target.result;
                const imageElement = <img src="${imageUrl}" loading="lazy" decoding="async">;

                preview_square.innerHTML = 
                    imageElement
            ;
            };

            // Lire le fichier en tant que donn√©es URL
            reader.readAsDataURL(imgInput.files[0]);
        } else {
            const imageElement = <img src="./media/aucune-image.png" loading="lazy" decoding="async">;

            preview_square.innerHTML = 
                    imageElement
        ;
        }
    }
